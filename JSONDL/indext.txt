function populateGroupFilter() {
    const groupSelect = document.getElementById('filterGroup');
    groupSelect.innerHTML = '<option value="">📚 Chọn nhóm đề thi</option>';
    
    examGroups.forEach(group => {
        // Count exams in this group
        const examCount = fullExamList.filter(exam => 
            exam.group === group.name
        ).length;
        
        if (examCount > 0) {
            const option = document.createElement('option');
            option.value = group.name;
            option.innerHTML = `
                ${group.icon || '📁'} ${group.name}
                <span class="exam-count-badge">${examCount}</span>
            `;
            option.dataset.color = group.color;
            option.dataset.description = group.description || '';
            groupSelect.appendChild(option);
        }
    });
    
    // Add "Tất cả" option
    const allOption = document.createElement('option');
    allOption.value = "__ALL__";
    allOption.textContent = "🌐 Tất cả đề thi";
    allOption.style.fontWeight = 'bold';
    groupSelect.appendChild(allOption);
    
    // Add event listener
    groupSelect.onchange = filterExamList;
}

        currentFilteredExams = [...fullExamList];

        currentFilteredExams = fullExamList.filter(exam => 
            exam.group === selectedGroup
        );
    }
    
    // Filter by current mode
    if (examMode === 'exam') {
        currentFilteredExams = currentFilteredExams.filter(exam => 
            exam.mode === 'exam' && exam.type !== 'theory_collection'
        );
    } else if (examMode === 'theory') {
        currentFilteredExams = currentFilteredExams.filter(exam => 
            exam.type === 'theory_collection'
        );
    } else {
        // Practice mode: show practice exams and theory
        currentFilteredExams = currentFilteredExams.filter(exam => 
            exam.mode === 'practice' || exam.type === 'theory_collection'
        );
    }
    
    // Sort by name
    currentFilteredExams.sort((a, b) => a.name.localeCompare(b.name));
    
    // Populate exam select
    currentFilteredExams.forEach(exam => {
        const option = document.createElement('option');
        option.value = exam.id;
        
        // Create badge based on type/mode
        let badge = '';
        if (exam.type === 'theory_collection') {
            badge = '<span style="color:#2196F3;"> [LÝ THUYẾT]</span>';
        } else if (exam.mode === 'exam') {
            badge = '<span style="color:#FF9800;"> [KIỂM TRA]</span>';
        } else {
            badge = '<span style="color:#4CAF50;"> [LUYỆN TẬP]</span>';
        }
        
        if (exam.question_count) {
            badge += ` <small style="color:#666;">(${exam.question_count} câu)</small>`;
        }
        
        option.innerHTML = `${exam.name}${badge}`;
        option.title = exam.description || exam.name;
        
        // Store all data in dataset
        Object.keys(exam).forEach(key => {
            if (typeof exam[key] !== 'object') {
                option.dataset[key] = exam[key];
            }
        });
        
        examSelect.appendChild(option);
    });
    
    // Update stats
    updateFilterStats(selectedGroup, currentFilteredExams.length);
    
    // Auto-select if only one exam
    if (currentFilteredExams.length === 1) {
        examSelect.value = currentFilteredExams[0].id;
        showNotification(`Tự động chọn đề duy nhất trong nhóm`, 'info');
    }
    
    // Add change event
    examSelect.onchange = handleExamSelection;
}

    }

// ============================================
// EXAM SELECTION HANDLING
// ============================================
async function handleExamSelection() {
    const selectedId = document.getElementById('selectExam').value;
    if (!selectedId) return;
    
    // Find selected exam
    const exam = fullExamList.find(e => e.id === selectedId);
    if (!exam) {
        showNotification('Không tìm thấy đề thi', 'error');
        return;
    }
    
    currentExam = exam;
    examSubmitted = false;
    
    // Show exam info
    showExamInfo(exam);
    
    // Check login requirement for exam mode
    if (exam.mode === 'exam' && !currentUser) {
        showNotification('🔒 Đề thi yêu cầu đăng nhập!', 'error');
        showModal('login-modal');
        document.getElementById('selectExam').value = "";
        return;
    }
    
    // Set exam mode
    if (exam.mode === 'exam') {
        examMode = 'exam';
        document.body.classList.add('exam-mode');
        document.body.classList.remove('practice-mode');
        showNotification('📝 Chế độ KIỂM TRA: Đã tắt xem đáp án', 'warning');
    } else {
        examMode = 'practice';
        document.body.classList.remove('exam-mode');
        document.body.classList.add('practice-mode');
    }
    
    try {
        showNotification(`📥 Đang tải đề: ${exam.name}...`, 'info');
        
        const response = await fetch(exam.file);
        if (!response.ok) throw new Error(`Không tải được file: ${exam.file}`);
        
        const data = await response.json();
        
        // Check if it's theory collection or regular exam
        if (exam.type === 'theory_collection') {
            currentTheoryData = data;
            renderTheoryCollection(data);
        } else {
            currentTheoryData = null;
            currentQuestions = Array.isArray(data) ? data : (data.questions || []);
            
            // Add default values if missing
            currentQuestions.forEach((q, idx) => {
                if (!q.id) q.id = `Q${idx + 1}`;
                if (!q.level) q.level = 'nhan_biet';
                if (!q.type) q.type = 'multiple_choice';
            });
            
            renderExamQuestions(currentQuestions);
        }
        
        document.title = `${exam.name} - Lớp Toán Thầy Bình`;
        showNotification(`✅ Đã tải thành công ${currentQuestions.length || 'nội dung'}`, 'success');
        
        // Save to history
        saveExamHistory(exam.id);
        
    } catch (error) {
        console.error("❌ Lỗi tải đề thi:", error);
        showNotification(`Lỗi: ${error.message}`, 'error');
        resetToWelcomeScreen();
    }
}

function renderExamQuestions(questions) {

                <h3 style="color: #dc3545;">⚠️ Không có câu hỏi nào</h3>

                    ← Quay lại trang chủ
                </button>
            </div>
        `;
        return;
    }
    
    let html = '';
    let currentPart = 0;
    
    questions.forEach((item, index) => {
        // Add part headers
        let newPart = 0;
        if (item.type === 'multiple_choice') newPart = 1;
        else if (item.type === 'true_false') newPart = 2;
        else if (item.type === 'short_answer') newPart = 3;
        
        if (newPart > currentPart) {
            currentPart = newPart;
            let partTitle = '', partDesc = '';
            
            if (currentPart === 1) {
                partTitle = 'PHẦN I. TRẮC NGHIỆM NHIỀU LỰA CHỌN';
                partDesc = 'Chọn một phương án đúng nhất cho mỗi câu hỏi.';
            } else if (currentPart === 2) {
                partTitle = 'PHẦN II. TRẮC NGHIỆM ĐÚNG SAI';
                partDesc = 'Xác định tính đúng/sai của các mệnh đề.';
            } else if (currentPart === 3) {
                partTitle = 'PHẦN III. TRẢ LỜI NGẮN';
                partDesc = 'Nhập đáp án ngắn cho mỗi câu hỏi.';
            }
            
            if (partTitle) {
                html += `
                    <div style="background: #e3f2fd; padding: 20px; border-radius: 12px; margin: 20px 0; border-left: 5px solid #2196F3;">
                        <h3 style="margin: 0; color: #0d47a1;">${partTitle}</h3>
                        <p style="margin: 10px 0 0 0; color: #546e7a; font-style: italic;">${partDesc}</p>
                    </div>
                `;
            }
        }
        
        // Render question based on type
        switch(item.type) {
            case 'multiple_choice':
                html += renderMultipleChoiceQuestion(item);
                break;
            case 'true_false':
                html += renderTrueFalseQuestion(item);
                break;
            case 'short_answer':
                html += renderShortAnswerQuestion(item);
                break;
            default:
                html += `<div class="question">Loại câu hỏi không hỗ trợ: ${item.type}</div>`;
        }
    });
    
    mainContent.innerHTML = html;
    
    // Initialize features
    setTimeout(() => {
        setupScrollTracking();
        createSlideList();
        if (window.MathJax) {
            if (MathJax.typeset) {
                MathJax.typeset();
            } else if (MathJax.typesetPromise) {
                MathJax.typesetPromise();
            }
        }
    }, 100);
}

function renderMultipleChoiceQuestion(item) {

        'nhan_biet': {class: 'nb', text: 'NB'},
        'thong_hieu': {class: 'th', text: 'TH'}, 
        'van_dung': {class: 'vd', text: 'VD'},
        'van_dung_cao': {class: 'vdc', text: 'VDC'}

    const chapterBadge = item.chapter ? `<span class="chapter-badge">${item.chapter}</span>` : '';

        const cleanValue = value.replace(/^[\$]?[A-Z]\.\s*/, '');

            </div>

    // Images
    let imagesHtml = '';
    if (item.images && Array.isArray(item.images) && item.images.length > 0) {

    } else if (item.image) {
        imagesHtml = `<img src="${item.image}" style="max-width: 100%; margin: 15px 0; border-radius: 8px;">`;
    }

    return `
        <div class="question" id="${item.id}">
            <div class="question-header">
                <div class="question-title">${item.title || `Câu hỏi`}</div>
                <span class="status-tag ${levelInfo.class}">${levelInfo.text}</span>
                ${chapterBadge}
            </div>

                    </div>

                    <div class="options">
                        ${optionsHtml}
                    </div>
                    
                    <div class="answer-check">

                    </div>

                            </div>
                        </div>

                </div>

    `;

function renderTrueFalseQuestion(item) {
    const levelMap = {
        'nhan_biet': {class: 'nb', text: 'NB'},
        'thong_hieu': {class: 'th', text: 'TH'},
        'van_dung': {class: 'vd', text: 'VD'}
    };
    const levelInfo = levelMap[item.level] || {class: 'nb', text: 'NB'};
    const chapterBadge = item.chapter ? `<span class="chapter-badge">${item.chapter}</span>` : '';

    // Images

    if (item.images && Array.isArray(item.images) && item.images.length > 0) {
        imagesHtml = `

                ${item.images.map(src => `<img src="${src}" style="max-width: 48%; border-radius: 8px;">`).join('')}
            </div>`;
    } else if (item.image) {
        imagesHtml = `<img src="${item.image}" style="max-width: 100%; margin: 15px 0; border-radius: 8px;">`;
    }

    // Statements
    let statementsHtml = '';
    if (Array.isArray(item.statements)) {
        item.statements.forEach((stmt, idx) => {
            const letter = String.fromCharCode(97 + idx);
            let stmtText = '';
            let stmtImage = '';

            if (typeof stmt === 'object' && stmt !== null) {
                stmtText = stmt.text || '';
                if (stmt.image) {
                    stmtImage = `<img src="${stmt.image}" style="max-height: 150px; border-radius: 5px; margin-top: 5px;">`;
                }
            } else {
                stmtText = stmt;
            }

            statementsHtml += `
                <div class="option-row">
                    <div style="flex: 1;">
                        <strong>${letter})</strong> ${processContent(stmtText)}
                        ${stmtImage}
                    </div>
                    <div style="display: flex; gap: 15px;">
                        <label style="display: flex; align-items: center; gap: 5px; cursor: pointer;">
                            <input type="radio" name="${item.id}_${letter}" value="True"
                                   onchange="handleAnswerChange('${item.id}_${letter}', 'True')">
                            <span style="color: #28a745; font-weight: bold;">Đúng</span>
                        </label>
                        <label style="display: flex; align-items: center; gap: 5px; cursor: pointer;">
                            <input type="radio" name="${item.id}_${letter}" value="False"

                            <span style="color: #dc3545; font-weight: bold;">Sai</span>
                        </label>
                    </div>
                </div>
            `;
        });
    }

    // Parse answers
    let answersArray = [];
    if (item.answers && typeof item.answers === 'string') {
        try {
            answersArray = JSON.parse(item.answers.replace(/'/g, '"'));
        } catch (e) {
            console.error("Parse answers error:", e);
        }
    } else if (Array.isArray(item.answers)) {
        answersArray = item.answers;
    }

    return `
        <div class="question" id="${item.id}">
            <div class="question-header">
                <div class="question-title">${item.title || `Câu hỏi Đúng/Sai`}</div>
                <span class="status-tag ${levelInfo.class}">${levelInfo.text}</span>
                ${chapterBadge}
            </div>
            
            <div style="font-size: 1.1em; line-height: 1.6; margin-bottom: 15px;">
                ${processContent(item.text || item.question || '')}
            </div>
            
            ${imagesHtml}
            
            <div class="options" style="margin-top: 20px;">
                ${statementsHtml}
            </div>
            
            <div class="answer-check">
                <button onclick="checkTrueFalseAnswer('${item.id}', ${JSON.stringify(answersArray)})">
                    ✅ Kiểm tra câu này
                </button>
                <div id="result-${item.id}" class="answer-result"></div>
            </div>
            
            ${item.solution && examMode !== 'exam' ? `
                <div class="secure-solution-container">
                    <button class="toggle-solution-btn" onclick="toggleSolution('${item.id}')">
                        👁️ Xem hướng dẫn giải
                    </button>
                    <div id="solution-${item.id}" class="solution-box">
                        <div class="solution-title">Hướng dẫn giải:</div>
                        <div>${processContent(item.solution)}</div>
                    </div>
                </div>
            ` : ''}
        </div>
    `;
}

window.checkAnswer = function(qid, correct) {
    if (examMode === 'exam' && !examSubmitted) {
        showNotification('Chỉ được kiểm tra đáp án sau khi nộp bài!', 'warning');
        return;
    }

    if (!container) return;

    if (!chosen) {
        showNotification("Hãy chọn một đáp án!", 'warning');
        return;
    }
    
    const chosenVal = chosen.value;
    const resultDiv = document.getElementById(`result-${qid}`);
    
    if (chosenVal === correct) {
        resultDiv.innerHTML = `
            <div class="answer-result correct">
                ✅ <strong>Chính xác!</strong><br>
                Đáp án của bạn: <b>${chosenVal}</b>
            </div>
        `;
        
        container.querySelectorAll('.option-row').forEach(row => {
            const radio = row.querySelector('input[type="radio"]');
            if (radio.value === correct) {
                row.style.background = '#d4edda';
                row.style.borderColor = '#28a745';
            }
        });
    } else {
        resultDiv.innerHTML = `
            <div class="answer-result incorrect">
                ❌ <strong>Sai rồi.</strong><br>
                Bạn chọn: <b>${chosenVal}</b><br>
                Đáp án đúng: <b style="color: #28a745">${correct}</b>
            </div>
        `;
        
        container.querySelectorAll('.option-row').forEach(row => {
            const radio = row.querySelector('input[type="radio"]');
            if (radio.value === correct) {
                row.style.background = '#d4edda';
                row.style.borderColor = '#28a745';
            } else if (radio.value === chosenVal) {
                row.style.background = '#f8d7da';
                row.style.borderColor = '#dc3545';
            }
        });
    }
};

window.checkTrueFalseAnswer = function(questionId, correctAnswers) {
    if (examMode === 'exam' && !examSubmitted) {
        showNotification('Chỉ được kiểm tra đáp án sau khi nộp bài!', 'warning');
        return;
    }
    
    const container = document.getElementById(questionId);
    if (!container) return;
    
    const resultDiv = document.getElementById(`result-${questionId}`);
    let html = '<div class="answer-result">';
    let correctCount = 0;
    
    if (Array.isArray(correctAnswers)) {
        for (let i = 0; i < 4; i++) {
            const letter = String.fromCharCode(97 + i);
            const radios = container.querySelectorAll(`input[name="${questionId}_${letter}"]`);
            const selected = Array.from(radios).find(r => r.checked);
            
            if (selected) {
                const isCorrect = (selected.value === 'True') === Boolean(correctAnswers[i]);
                if (isCorrect) correctCount++;
                
                html += `
                    ${letter}) ${isCorrect ? '✅' : '❌'} 
                    (Chọn: ${selected.value === 'True' ? 'Đúng' : 'Sai'}, 
                    Đáp án: ${correctAnswers[i] ? 'Đúng' : 'Sai'})<br>
                `;
            } else {
                html += `${letter}) ⭕ Chưa chọn<br>`;
            }
        }
        
        const score = correctCount * 0.25;
        html += `<br><strong>Điểm: ${score.toFixed(2)}/1.00</strong>`;
    }
    
    html += '</div>';
    resultDiv.innerHTML = html;
};

window.calculateFinalScore = async function() {
    if (!currentExam) {
        showNotification('Vui lòng chọn đề thi trước!', 'error');
        return;
    }
    
    if (examMode === 'exam' && !currentUser) {
        showNotification('Vui lòng đăng nhập để nộp bài!', 'error');
        showModal('login-modal');
        return;
    }
    
    // Collect answers
    const studentAnswers = {};
    const correctAnswers = {};
    let totalQuestions = 0;
    let answeredQuestions = 0;
    
    // Collect data from questions
    if (currentQuestions && currentQuestions.length > 0) {
        currentQuestions.forEach(q => {
            totalQuestions++;
            
            // Get correct answer
            if (q.type === 'multiple_choice') {
                correctAnswers[q.id] = q.correct;
            } else if (q.type === 'true_false') {
                correctAnswers[q.id] = q.answers;
            } else if (q.type === 'short_answer') {
                correctAnswers[q.id] = q.answer;
            }
            
            // Get student answer
            let userAnswer = null;
            if (q.type === 'multiple_choice') {
                const selected = document.querySelector(`input[name="${q.id}"]:checked`);
                userAnswer = selected ? selected.value : null;
            } else if (q.type === 'true_false') {
                // Collect all 4 answers
                userAnswer = [];
                for (let i = 0; i < 4; i++) {
                    const letter = String.fromCharCode(97 + i);
                    const selected = document.querySelector(`input[name="${q.id}_${letter}"]:checked`);
                    userAnswer.push(selected ? (selected.value === 'True') : null);
                }
            } else if (q.type === 'short_answer') {
                const input = document.getElementById(`input-${q.id}`);
                userAnswer = input ? input.value.trim() : null;
            }
            
            studentAnswers[q.id] = userAnswer;
            if (userAnswer !== null && userAnswer !== '') {
                if (Array.isArray(userAnswer)) {
                    if (userAnswer.some(ans => ans !== null)) {
                        answeredQuestions++;
                    }
                } else {
                    answeredQuestions++;
                }
            }
        });
    }
    
    // Calculate score
    let totalScore = 0;
    let maxScore = 0;
    
    if (currentQuestions && currentQuestions.length > 0) {
        currentQuestions.forEach(q => {
            const userAnswer = studentAnswers[q.id];
            const correctAnswer = correctAnswers[q.id];
            
            // Calculate max possible score for this question
            let questionMaxScore = 0;
            if (q.type === 'multiple_choice') questionMaxScore = 0.25;
            else if (q.type === 'true_false') questionMaxScore = 1.00;
            else if (q.type === 'short_answer') questionMaxScore = 0.5;
            
            maxScore += questionMaxScore;
            
            // Calculate actual score
            if (userAnswer !== null && userAnswer !== '') {
                if (q.type === 'multiple_choice') {
                    if (String(userAnswer).toUpperCase() === String(correctAnswer).toUpperCase()) {
                        totalScore += 0.25;
                    }
                } else if (q.type === 'true_false' && Array.isArray(userAnswer) && Array.isArray(correctAnswer)) {
                    let correctCount = 0;
                    for (let i = 0; i < Math.min(userAnswer.length, correctAnswer.length); i++) {
                        if (userAnswer[i] !== null && Boolean(userAnswer[i]) === Boolean(correctAnswer[i])) {
                            correctCount++;
                        }
                    }
                    // Calculate partial score
                    if (correctCount === 1) totalScore += 0.1;
                    else if (correctCount === 2) totalScore += 0.25;
                    else if (correctCount === 3) totalScore += 0.5;
                    else if (correctCount === 4) totalScore += 1.0;
                } else if (q.type === 'short_answer') {
                    const normalizedUser = String(userAnswer).toLowerCase().replace(/\s+/g, '');
                    const normalizedCorrect = String(correctAnswer).toLowerCase().replace(/\s+/g, '');
                    if (normalizedUser === normalizedCorrect) {
                        totalScore += 0.5;
                    }
                }
            }
        });
    }
    
    // Format score
    totalScore = Math.round(totalScore * 100) / 100;
    if (maxScore > 10) {
        // Scale to 10
        totalScore = (totalScore / maxScore) * 10;
        totalScore = Math.round(totalScore * 100) / 100;
    }
    
    // Mark as submitted for exam mode
    if (examMode === 'exam') {
        examSubmitted = true;
        document.body.classList.add('exam-submitted');
        document.body.classList.remove('exam-mode');
        showNotification('✅ Đã nộp bài! Bạn có thể xem đáp án và lời giải.', 'success');
    }
    
    // Show results
    showScoreResult(totalScore, answeredQuestions, totalQuestions, studentAnswers, correctAnswers);
    
    // Save attempt if logged in
    if (currentUser && currentExam) {
        await saveAttempt(totalScore, studentAnswers, correctAnswers);
    }
};

function createSlideList() {
    const slideList = document.getElementById('slideList');
    if (!slideList) return;
    
    let html = '';
    
    // For exam questions
    if (currentQuestions && currentQuestions.length > 0) {
        currentQuestions.forEach((q, index) => {
            const status = getQuestionStatus(q.id);
            const statusIcon = status === 'correct' ? '✅' : 
                             status === 'incorrect' ? '❌' : 
                             status === 'answered' ? '⭕' : '⬜';
            
            html += `
                <div class="slide-item" data-index="${index}">
                    <span class="slide-number">${index + 1}</span>
                    <span class="slide-status">${statusIcon}</span>
                    <span class="slide-title">${q.title || `Câu ${index + 1}`}</span>
                </div>
            `;
        });
    } else if (currentTheoryData && currentTheoryData.sections) {
        // For theory sections
        currentTheoryData.sections.forEach((section, index) => {
            html += `
                <div class="slide-item" data-index="${index}">
                    <span class="slide-number">${index + 1}</span>
                    <span class="slide-title">${section.title || `Chương ${index + 1}`}</span>
                </div>
            `;
        });
    } else {
        html = '<div class="slide-item">Chưa có nội dung</div>';
    }
    
    slideList.innerHTML = html;
    
    // Add click events
    slideList.querySelectorAll('.slide-item[data-index]').forEach(item => {
        item.addEventListener('click', () => {
            const index = parseInt(item.dataset.index);
            scrollToSection(index);
            toggleSlideMenu();
        });
    });
}

function updateSlideList() {
    const slideList = document.getElementById('slideList');
    if (!slideList) return;
    
    slideList.querySelectorAll('.slide-item').forEach((item, index) => {
        if (currentQuestions && currentQuestions[index]) {
            const q = currentQuestions[index];
            const status = getQuestionStatus(q.id);
            const statusIcon = status === 'correct' ? '✅' : 
                             status === 'incorrect' ? '❌' : 
                             status === 'answered' ? '⭕' : '⬜';
            
            const statusSpan = item.querySelector('.slide-status');
            if (statusSpan) {
                statusSpan.textContent = statusIcon;
            }
            
            // Update colors
            if (status === 'correct') {
                item.style.background = '#e8f5e9';
                item.style.borderLeftColor = '#28a745';
            } else if (status === 'incorrect') {
                item.style.background = '#ffebee';
                item.style.borderLeftColor = '#dc3545';
            } else if (status === 'answered') {
                item.style.background = '#fff3e0';
                item.style.borderLeftColor = '#ffc107';
            }
        }
    });
}

function getQuestionStatus(qId) {
    const userAnswer = window.studentAnswers[qId];
    
    if (userAnswer === undefined || userAnswer === null || userAnswer === '') {
        return 'unanswered';
    }
    
    if (Array.isArray(userAnswer) && userAnswer.every(ans => ans === null)) {
        return 'unanswered';
    }
    
    // For exam mode before submission, don't show correctness
    if (examMode === 'exam' && !examSubmitted) {
        return 'answered';
    }
    
    // Check correctness if we have correct answer
    const question = currentQuestions?.find(q => q.id === qId);
    if (!question) return 'answered';
    
    const correctAnswer = question.correct || question.answer || question.answers;
    
    if (question.type === 'multiple_choice') {
        return String(userAnswer).toUpperCase() === String(correctAnswer).toUpperCase() ? 'correct' : 'incorrect';
    } else if (question.type === 'short_answer') {
        const normUser = String(userAnswer).toLowerCase().replace(/\s+/g, '');
        const normCorrect = String(correctAnswer).toLowerCase().replace(/\s+/g, '');
        return normUser === normCorrect ? 'correct' : 'incorrect';
    }
    
    return 'answered';
}

function scrollToSection(index) {
    if (currentQuestions && currentQuestions.length > index) {
        const element = document.getElementById(currentQuestions[index].id);
        if (element) {
            element.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
    } else if (currentTheoryData && currentTheoryData.sections && currentTheoryData.sections.length > index) {
        const element = document.getElementById(`section-${index}`);
        if (element) {
            element.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
    }
}

function showModal(modalId) {
    const modal = document.getElementById(modalId);
    if (modal) {
        modal.classList.add('active');
        activeModal = modalId;
        document.body.style.overflow = 'hidden'; // Prevent scrolling
    }
}

function hideModal(modalId) {
    const modal = document.getElementById(modalId);
    if (modal) {
        modal.classList.remove('active');
        if (activeModal === modalId) {
            activeModal = null;
        }
        document.body.style.overflow = ''; // Restore scrolling
    }
}

function setupAuthentication() {
    const loginForm = document.getElementById('code-login-form');
    if (!loginForm) return;
    
    loginForm.addEventListener('submit', async function(e) {
        e.preventDefault();
        
        const code = document.getElementById('login-code').value.trim();
        const name = document.getElementById('student-name').value.trim() || 'Học sinh';
        
        if (!code) {
            showNotification('Vui lòng nhập mã đăng nhập', 'error');
            return;
        }
        
        try {
            if (code === 'admin79') {
                // Teacher login
                await loginAsTeacher(name);
            } else if (code.startsWith('HS') || code.startsWith('GV')) {
                // Student login via Firebase
                await loginAsStudent(code, name);
            } else {
                showNotification('Mã đăng nhập không hợp lệ', 'error');
            }
        } catch (error) {
            console.error('Login error:', error);
            showNotification('Lỗi đăng nhập: ' + error.message, 'error');
        }
    });
}

async function loginAsTeacher(name) {
    isTeacherLoggedIn = true;
    currentUser = {
        uid: 'teacher_' + Date.now(),
        name: name,
        role: 'teacher',
        code: 'admin79'
    };
    
    // Update UI
    document.getElementById('loginBtn').innerHTML = `👨‍🏫 ${name}`;
    document.getElementById('loginBtn').style.background = '#28a745';
    
    // Show search box
    const searchBox = document.getElementById('student-search-box');
    if (searchBox) searchBox.classList.add('active');
    
    hideModal('login-modal');
    showNotification(`✅ Đăng nhập thành công với quyền Giáo viên: ${name}`, 'success');
    
    // Switch to exam mode if requested
    if (examMode === 'exam') {
        filterExamList();
        showNotification('📝 Chuyển sang chế độ KIỂM TRA với quyền giáo viên', 'info');
    }
}

async function loginAsStudent(code, name) {
    try {
        // Check if student code exists in Firebase
        const snapshot = await db.ref(`studentCodes/${code}`).once('value');
        const studentData = snapshot.val();
        
        if (!studentData) {
            showNotification('Mã học sinh không tồn tại!', 'error');
            return;
        }
        
        if (studentData.name !== name) {
            showNotification('Họ tên không khớp với mã học sinh!', 'error');
            return;
        }
        
        // Login successful
        isTeacherLoggedIn = false;
        currentUser = {
            uid: 'student_' + code,
            name: name,
            role: 'student',
            code: code,
            classCode: studentData.classCode,
            className: studentData.className
        };
        
        // Update last login
        await db.ref(`studentCodes/${code}`).update({
            lastLogin: firebase.database.ServerValue.TIMESTAMP
        });
        
        // Save to users
        await db.ref(`users/${currentUser.uid}`).set({
            name: name,
            role: 'student',
            code: code,
            lastLogin: firebase.database.ServerValue.TIMESTAMP
        });
        
        // Update UI
        document.getElementById('loginBtn').innerHTML = `👤 ${name}`;
        const searchBox = document.getElementById('student-search-box');
        if (searchBox) searchBox.classList.add('active');
        
        hideModal('login-modal');
        showNotification(`✅ Đăng nhập thành công: ${name} (${code})`, 'success');
        
        // Ask for session code
        if (studentData.classCode) {
            const joinSession = confirm(`Bạn thuộc lớp ${studentData.className} (${studentData.classCode}). Tham gia lớp học?`);
            if (joinSession) {
                sessionCode = studentData.classCode;
                showNotification(`✅ Đã tham gia lớp học: ${studentData.className}`, 'success');
            }
        }
        
        // Switch to exam mode if requested
        if (examMode === 'exam') {
            filterExamList();
            showNotification('📝 Chuyển sang chế độ KIỂM TRA với tài khoản học sinh', 'info');
        }
        
    } catch (error) {
        console.error('Student login error:', error);
        showNotification('Lỗi đăng nhập học sinh: ' + error.message, 'error');
    }
}

async function saveAutoScore(questionId, answer) {
    if (!currentUser || !currentExam || !sessionCode) return;
    
    try {
        await db.ref(`auto_scores/${sessionCode}/${currentUser.uid}/${currentExam.id}/${questionId}`).set({
            answer: answer,
            timestamp: firebase.database.ServerValue.TIMESTAMP
        });
    } catch (error) {
        console.error('Error saving auto score:', error);
    }
}

async function saveAttempt(score, studentAnswers, correctAnswers) {
    if (!currentUser || !currentExam) return;
    
    try {
        const attemptId = `${currentExam.id}_${currentUser.code}_${Date.now()}`;
        const attemptData = {
            attemptId: attemptId,
            studentCode: currentUser.code,
            studentName: currentUser.name,
            examId: currentExam.id,
            examName: currentExam.name,
            totalScore: parseFloat(score),
            totalQuestions: currentQuestions ? currentQuestions.length : 0,
            answeredQuestions: Object.keys(studentAnswers).filter(k => 
                studentAnswers[k] && studentAnswers[k] !== '' && 
                !(Array.isArray(studentAnswers[k]) && studentAnswers[k].every(item => item === null))
            ).length,
            studentAnswers: studentAnswers,
            correctAnswers: correctAnswers,
            timestamp: firebase.database.ServerValue.TIMESTAMP,
            mode: examMode
        };
        
        // Save to attempts
        await db.ref(`attempts/${attemptId}`).set(attemptData);
        
        // Save to student's history
        await db.ref(`students/${currentUser.code}/attempts/${attemptId}`).set({
            examId: currentExam.id,
            examName: currentExam.name,
            score: parseFloat(score),
            timestamp: firebase.database.ServerValue.TIMESTAMP
        });
        
        console.log(`✅ Saved attempt ${attemptId} with score ${score}`);
        
    } catch (error) {
        console.error('Error saving attempt:', error);
    }
}

function saveExamHistory(examId) {
    if (!examId) return;
    
    try {
        const history = JSON.parse(localStorage.getItem('examHistory') || '[]');
        const exam = fullExamList.find(e => e.id === examId);
        
        if (exam) {
            // Remove if already exists
            const existingIndex = history.findIndex(h => h.id === examId);
            if (existingIndex !== -1) {
                history.splice(existingIndex, 1);
            }
            
            // Add to beginning
            history.unshift({
                id: exam.id,
                name: exam.name,
                timestamp: Date.now(),
                mode: examMode
            });
            
            // Keep only last 20
            if (history.length > 20) {
                history.pop();
            }
            
            localStorage.setItem('examHistory', JSON.stringify(history));
        }
    } catch (error) {
        console.error('Error saving history:', error);
    }
}

function showRecentExams() {

        if (history.length > 0) {
            console.log('Recent exams:', history);
        }
    } catch (error) {
        console.error('Error loading history:', error);
    }
}

// SCROLL TRACKING
// ============================================
function setupScrollTracking() {
    const pageContainer = document.getElementById('main-content');
    if (!pageContainer) return;
    
    // Scroll progress
    pageContainer.addEventListener('scroll', updateScrollProgress);
    
    // Intersection Observer for active sections
    scrollObserver = new IntersectionObserver(
        (entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    // Update active section in menu
                    const sections = document.querySelectorAll('.question, .theory-section');
                    const index = Array.from(sections).indexOf(entry.target);
                    
                    if (index !== -1) {
                        updateActiveSlideInMenu(index);
                    }
                }
            });
        },
        {
            root: pageContainer,
            rootMargin: '-20% 0px -60% 0px',
            threshold: 0.1
        }
    );
    
    // Observe all questions and sections
    document.querySelectorAll('.question, .theory-section').forEach(el => {
        scrollObserver.observe(el);
    });
}

function updateScrollProgress() {
    const pageContainer = document.getElementById('main-content');
    if (!pageContainer) return;
    
    const scrollTop = pageContainer.scrollTop;
    const scrollHeight = pageContainer.scrollHeight - pageContainer.clientHeight;
    const scrollPercentage = (scrollTop / scrollHeight) * 100;
    
    const progressBar = document.getElementById('scrollProgress');
    if (progressBar) {
        progressBar.style.width = scrollPercentage + '%';
    }
}

function updateActiveSlideInMenu(index) {
    const slideItems = document.querySelectorAll('.slide-item');
    slideItems.forEach((item, i) => {
        if (i === index) {
            item.classList.add('active');
        } else {
            item.classList.remove('active');
        }
    });
}

function setupFullscreen() {
    const fullscreenBtn = document.getElementById('fullscreenBtn');
    if (!fullscreenBtn) return;
    
    fullscreenBtn.addEventListener('click', toggleFullscreen);
    document.addEventListener('fullscreenchange', updateFullscreenButton);
    document.addEventListener('keydown', function(e) {
        if (e.key === 'F11') {
            e.preventDefault();
            toggleFullscreen();
        }
    });
}

function toggleFullscreen() {
    if (!document.fullscreenElement) {
        if (document.documentElement.requestFullscreen) {
            document.documentElement.requestFullscreen();
        }
    } else {
        if (document.exitFullscreen) {
            document.exitFullscreen();
        }
    }
}

function updateFullscreenButton() {
    const btn = document.getElementById('fullscreenBtn');
    if (btn) {
        btn.textContent = document.fullscreenElement ? '📺 Thoát' : '📺 Fullscreen';
    }
}

// CONTENT PROCESSING
// ============================================
function processContent(text) {
    if (!text || typeof text !== 'string') return '';
    
    // Protect LaTeX tables
    const placeholders = [];
    const tableRegex = /\\begin\{tabular\}\s*\{([^}]+)\}([\s\S]*?)\\end\{tabular\}/g;
    
    text = text.replace(tableRegex, function(match, colSpecs, content) {
        const htmlTable = convertTableToHTML(colSpecs, content);
        placeholders.push(htmlTable);
        return `___TABLE_PLACEHOLDER_${placeholders.length - 1}___`;
    });
    
    // Protect math expressions
    text = text.replace(/\$([^$]+)\$/g, function(match, mathContent) {
        let safeMath = mathContent
            .replace(/</g, ' \\lt ')
            .replace(/>/g, ' \\gt ')
            .replace(/&/g, ' \\& ');
        return '$' + safeMath + '$';
    });
    
    // HTML entities
    text = text.replace(/&(?!(?:amp|lt|gt|quot|#39);)/g, '&amp;')
               .replace(/</g, '&lt;')
               .replace(/>/g, '&gt;');
    
    // Restore tables
    placeholders.forEach((htmlTable, index) => {
        text = text.replace(`___TABLE_PLACEHOLDER_${index}___`, htmlTable);
    });
    
    return text;
}

function convertTableToHTML(colSpecs, content) {
    try {
        let cleanContent = content
            .replace(/\\hline/g, '')
            .replace(/\\cline\{[^}]*\}/g, '')
            .trim();
            
        let rows = cleanContent.split('\\\\').filter(r => r.trim() !== '');
        
        let htmlRows = rows.map(row => {
            let cells = [];
            let currentCell = '';
            let inMath = false;
            let braceCount = 0;
            
            for (let i = 0; i < row.length; i++) {
                const char = row[i];
                
                if (char === '\\' && row[i+1] === '$') { 
                    currentCell += '\\$'; i++; continue;
                }

                if (char === '$') {
                    inMath = !inMath;
                    currentCell += char;
                } else if (char === '{') {
                    braceCount++;
                    currentCell += char;
                } else if (char === '}') {
                    braceCount--;
                    if (braceCount < 0) braceCount = 0;
                    currentCell += char;
                } else if (char === '&' && !inMath && braceCount === 0) {
                    cells.push(currentCell);
                    currentCell = '';
                } else {
                    currentCell += char;
                }
            }
            cells.push(currentCell);
            
            let htmlCells = cells.map(cell => {
                let c = cell.trim();
                c = c.replace(/\\multicolumn\{[^}]*\}\{[^}]*\}\{([^}]*)\}/g, '$1')
                     .replace(/\\textbf\{([^}]*)\}/g, '<strong>$1</strong>');
                
                return `<td style="border: 1px solid #ddd; padding: 8px;">${c}</td>`;
            }).join('');
            
            return `<tr>${htmlCells}</tr>`;
        }).join('');

        return `<div style="overflow-x: auto; margin: 15px 0;">
                    <table style="width: 100%; border-collapse: collapse;">
                        ${htmlRows}
                    </table>
                </div>`;
    } catch (e) {
        console.error("Lỗi convert bảng:", e);
        return `<div style="color:red">[Lỗi bảng]</div>`;
    }
}
